namespace ConsoleApp1
{
    internal class Program
    {
        static void Main(string[] args)
        {
            bool isWork = true;
            List<string> notes = new List<string>();
            while (isWork)
            {
                ShowList(notes);
                CalledNotes();
                Console.SetCursorPosition(0, 0);

                int choiseOperstion = ChekedNumber("Введите номер операции: ");
                switch (choiseOperstion)
                {
                    case 1:
                        addList(ref notes);
                        break;
                    case 2:
                        reverseList(notes);
                        break;
                    case 3:
                        ClearList(ref notes);
                        break;
                    case 4:
                        SortList(notes);
                        break;
                    case 5:
                        isWork = false;
                        break;
                    case 6:
                        DeleteConcertin(ref notes);
                        break;
                }
                if (isWork)
                {
                    Console.WriteLine("Введите Enter: ");
                    Console.ReadLine();
                }
                Console.Clear();
            }
        }
        private static void CalledNotes()
        {
            Console.SetCursorPosition(0, 20);
            menu();
        }
        private static void ShowList(List<string> notes)
        {
            for (int i = 0; i < notes.Count; i++)
            {
                Console.SetCursorPosition(30, i);
                Console.WriteLine((i + 1) + " " + notes[i]);
            }
        }
        private static void addList(ref List<string> notes)
        {
            string input = Console.ReadLine();
            notes.Add(input);
        }
        private static void ClearList(ref List<string> notes)
        {
            notes.Clear();
        }
        private static void SortList(List<string> notes)
        {
            List<string> notesClone = new List<string>(notes);
            notesClone.Sort();
            for (int i = 0; i < notesClone.Count; i++)
            {
                Console.WriteLine(((notesClone.Count - i)) + " " + notesClone[i]);
            }
        }
        private static void DeleteConcertin(ref List<string> notes)
        {
            int notesNumber = ChekedNumber("Введите номер заметки: ");
            if (notesNumber > 0 && notesNumber <= notes.Count)
            {
                notes.RemoveAt(notesNumber - 1);
            }
        }
        private static void reverseList(List<string> notes)
        {
            List<string> notesReverseClone = new List<string>(notes);

            notesReverseClone.Reverse();
            for (int i = 0; i < notesReverseClone.Count; i++)
            {
                Console.WriteLine(((notesReverseClone.Count - i)) + " " + notesReverseClone[i]);
            }
        }
        private static void menu()
        {
            Console.WriteLine("1 - добавить заметку");
            Console.WriteLine("2 - показать в обратном порядку");
            Console.WriteLine("3 - очистить заметки");
            Console.WriteLine("4 - отсортировать в алфавитном порядке");
            Console.WriteLine("5 - выход");
            Console.WriteLine("6 - удаление конркетной заметки");
        }
        private static int ChekedNumber(string massage)
        {
            int num = 0;
            while (true)
            {
                Console.Write(massage);
                string input = Console.ReadLine();
                if (int.TryParse(input, out int resul))
                {
                    num = resul;
                    break;
                }
                Console.WriteLine("Некоректный ввод");
            }
            return num;
        }
    }
}
